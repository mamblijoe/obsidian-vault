>[!info] Замена сложных выражений на обращение к таблице подстановки
```ts
// Хуже
if (type === "primitive") {
  if (module === "number") {
    if ((entity = "int")) {
      return Primitive.number.int();
    }
  }
}

// Лучше
const commands = {
  primitive: {
    number: {
      int: () => {
        // ...
      },
    },
  },
};

const result = commands[type][module][entity];
```
